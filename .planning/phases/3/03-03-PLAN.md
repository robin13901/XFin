```dart
/// Pure utility functions for timeframe/date-integer calculations.
/// Extracted from AnalysisDao for independent testability and reuse.

/// Returns (startOfMonth, endOfMonth) as yyyyMMdd integers for the given [date].
(int, int) getMonthStartEnd(DateTime date) {
  final startOfMonth = date.year * 10000 + date.month * 100 + 1;
  final lastDayOfMonth = DateTime(date.year, date.month + 1, 0);
  final endOfMonth = lastDayOfMonth.year * 10000 +
      lastDayOfMonth.month * 100 +
      lastDayOfMonth.day;
  return (startOfMonth, endOfMonth);
}

/// Returns the intersection of [date]'s month boundaries with [filterStart]..[filterEnd],
/// or null if they don't overlap. All values are yyyyMMdd integers.
(int, int)? getMonthTimeframeIntersection(
  DateTime date, {
  required int filterStart,
  required int filterEnd,
}) {
  final (monthStart, monthEnd) = getMonthStartEnd(date);
  final effectiveStart = monthStart > filterStart ? monthStart : filterStart;
  final effectiveEnd = monthEnd < filterEnd ? monthEnd : filterEnd;
  if (effectiveStart > effectiveEnd) return null;
  return (effectiveStart, effectiveEnd);
}

/// Converts a yyyyMMdd date-int to a yyyyMMddHHmmss datetime-int (start of day: 000000).
int monthStartDateTimeInt(int startDate) => startDate * 1000000;

/// Converts a yyyyMMdd date-int to a yyyyMMddHHmmss datetime-int (end of day: 235959).
int monthEndDateTimeInt(int endDate) => endDate * 1000000 + 235959;
```

**CRITICAL DESIGN DECISION:** `getMonthTimeframeIntersection` takes `filterStart` and `filterEnd` as explicit parameters instead of reading from `global_constants.dart` globals. This makes the function pure and testable. The DAO will pass the current global values when calling it.
